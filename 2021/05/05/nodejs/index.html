1.3 nodejs简单介绍

1.3.1 nodejs简单介绍

  • 每个文件都是一个模块

    函数执行时，传入5个实参，当node在执行模块的代码时，它会首先在代码的最顶部，添加如下代码

    var module = function(exports, require, module, __filename, __dirname){..                           return module.exports                                                 }
// module.exports 直接导出，而非挂载
    exports: 该对象用来讲变量或函数暴露给外部
require: 函数，用来引入外部的模块
module: -module 代表的是当前模块本身
		-exports 就是module的属性
       (注：module.exports == exports，exports引用了module.exports)
__filename: 动态获取当前模块完整路径
__dirname: 动态获取当前模块所在文件夹的绝对路径
    console.log(global) 注: node中有全局变量global类似网页中的window，在全局中创建的变量都会作为global的属性保存，在全局中创建的函数都会作为global的方法保存

console.log(arguments) 注: 伪数组，用于存放函数的实参

2. nodejs的核心模块

2.1 文件模块(File System)

  • 所有操作都有同步、异步两种形式可供选择，带Sync为同步，不带Sync为异步

    • 同步文件系统会阻塞程序的执行，也就是除非操作完毕，否则不会向下执行代码
    • 异步文件系统不会阻塞程序的执行，而是在操作完成时，通过回调函数将结果返回

  • 文件的写入

    • 打开文件

      fs.open(path, flags[, mode], callback(err, fd))
// fd : 文件的描述符
// flags : 要打开文件要做的操作的类型
// mode : 设置文件的操作权限，一般不传

    • 向文件写入内容

      fs.write(fd, 'text',callback(err) )

    • 保存并关闭文件

      fs.close(fd, callback(err))

  • fs 其他的方法

    fs.existsSync(path)  -检查一个文件是否存在
fs.stat(path, callback) -获取文件信息 会返回一个对象stats，这个对象中保存了当前对象状态的相关信息
//stats.isFile() : 是否是一个文件
//stats.isDirectory() : 是否是一个文件夹
//stats.size : 文件大小
fs.unlink(path, callback) -删除文件
fs.readdir(path[, options], callback) -读取一个目录的目录结构
fs.truncate(path, len, callback) -截断文件， 截取单位为字节
fs.mkdir(path[, mode], callback) -创建一个目录(文件夹)
fs.rmdir(path, callback) -删除一个目录
fs.rename(oldPath, newPath, callback) -对文件夹进行重命名(剪切)
fs.watchFile(filename[, options], listener(curr, prev)) -监视文件的修改
// options : { interval: 1000} interval : 检查时间的间隔
// listener : 回调函数，当文件发生变化时，回调函数会执行
// curr 当前文件状态， prev 修改前文件状态

3.2 promise fs API


var fs = require("fs");

const pReadFile = function (filePath) {
  return new Promise((resolve, reject) => {
    fs.readFile(filePath, "utf8", (err, data) => {
      if (err) {
        reject(err);
      } else {
        resolve(data);
      }
    })
  })
}

3.3 封装Ajax 请求

function get(url, callback) {
      var oReq = new XMLHttpRequest();
      // 当请求加载成功之后要调用指定的函数
      oReq.onload = () => {
        callback(oReq.responseText)
      };
      oReq.open("get", url, true);
      oReq.send();
    }
    get('data.json', (data) => {
      console.log(data);
    })

3.3 promise 基于Ajax的应用场景

function get(url, callback) {
      return new Promise((resolve, reject) => {
          var oReq = new XMLHttpRequest();
      	  // 当请求加载成功之后要调用指定的函数
      	  oReq.onload = () => {
          callback && callback(oReq.responseText)
       	  resolve(oReq.responseText)
      	  };
          oReq.onerror = (err) => {
              reject(err)
          }
      	  oReq.open("get", url, true);
      	  oReq.send();
      })
    }
get('data.json', (data) => {
      console.log(data);
}
get('data.json')
	.then(res => {
    	console(res)
	}, err => {
    	console(err)

4. Express

4.1 Express的安装与使用

//0 安装
npm install express --save
//1 引包
var express = require('express')
//2 创建你服务器应用程序
var app = express()  // => http.createServer

//当服务器收到get请求时，执行回调函数
 app.get('/', (req, res) => { res.send('Hello Express!')}) // get => req.query
//公开指定目录
app.use('/public/', express.static('./public/'))
//忽略第一个参数的时候，则必须通过忽略/public的方式来使用，不省略则无法使用
app.use('/a/', express.static('./public/'))
// 即 a 相当于 public 的别名

4.2 扩展 关于文件模块路径，跟模块加载路径

4.2.1 文件路径

  • 文件操作路径中，相对路径设计的就是相对于执行node命令所处的路径,可以通过___dirname和path.join获得绝对路径，模块路径则不用担心
'./data/a.txt'  相对路径
'data/a.txt'  	相对路径
'/data/a.txt'   相对路径
'c:/data/a.txt' 相对路径

4.2.2模块加载路径

'./data/a.txt'  相对路径
'data/a.txt'  	Error: Cannot find module 'data/a.txt'
'/data/a.txt'   相对路径
'c:/data/a.txt' 相对路径

4.2.3 关于服务器中的路径

  • 哪些资源可以被用户访问，哪些不可以被用户访问，可以由代码决定
  • 在服务端中，文件中的路径就不要去写相对路径了，因为这个时候所有的资源都是通过url标识来获取的，开放哪个目录，请求路径就写成那个目录
  • 例 ： 开放/public/ 目录，则请求路径为/public/xxx

4.10 中间件

注意： 同一个请求，所经历的请求对象跟响应对象都是同一个

4.10.1应用级别的中间件

万能匹配
app.use((req, res, next) => {
  console('Time'),
  next()
})
只要/xxx开头的
app.use('/xxx', (req, res, next) => {
  console('Time'),
  next()
})

4.10.2 路由级别的中间件

get
app.get('/', (req, res) => {
	console('Time')
});
post
app.post('/', (req, res) => {
	console('Time')
});
put
app.put('/', (req, res) => {
	console('Time')
});
delete
app.delete('/', (req, res) => {
	console('Time')
});

4.10.3 Express中，对中间件的分类

  • 不关心请求路径和请求方法的中间件，任何请求均可进入
  • 中间件本身是一个方法，该方法接受三个参数
    • Request：请求对象
    • Response: 响应对象
    • next：进入下一个中间件

注：当一个请求进入到一个中间件，不调用next方法，则会停留在这个中间件中

​ 如果没有匹配的路径，则Express会默认输出Cannot GET 路径

6.3 Cookie 和 Session 的区别

http是无状态的。

  • Cookie, 存储在客户端上，是服务器给客户端添加的内容，不太安全，可以用来保存一些不太敏感的数据
  • Session,存储在服务器上，比较安全，用来保存状态，类似保险柜，发送客服端一把钥匙(Cookie)，钥匙丢失，状态丢失
